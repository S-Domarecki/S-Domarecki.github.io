Go was designed at Google in 2007 to improve programming productivity 
in an era of multicore, networked machines and large codebases.[21] 
The designers wanted to address criticism of other languages in use at Google, 
but keep their useful characteristics:[22]
* Static typing and run-time efficiency (like C)
* Readability and usability (like Python or JavaScript)[23]

Cechy języka:

  • statycznie typowany, kompilowany, wieloparadygmatowy
  • współbieżność na poziomie słowa kluczowego

Error & defer

package main

import (
    "fmt"
    "os" // grupowanie importów
)

func main() {
    // obsługa błędów przyklejona do obsługi rezultatu
    file, err := os.Open("example.txt") 
    if err != nil {
        fmt.Println("Error opening file:", err)
        return
    }
    defer file.Close() // wykonany na końcu

    data := make([]byte, 100) // statyczna wielkość
    count, err := file.Read(data)
    if err != nil {
        fmt.Println("Error reading file:", err)
        return
    }

    fmt.Println("Read", count, "bytes from file:", string(data))
}

Kawałek proda

func (mgr *PrometheusAlertManager) ProcessBatches(ctx context.Context, alertBatches *[]Alert.PrometheusAlertWrapper, dispatcher *dispatchers.Dispatcher) {
	var span trace.Span
	ctx, span = mgr.tracer.Start(ctx, "Batches processing")
    defer span.End()

	var wg sync.WaitGroup
	wg.Add(len(*alertBatches))
    defer wg.Wait()

	for i := range *alertBatches {
		go func(i int) {
			defer wg.Done()
			mgr.processBatch(ctx, i, &(*alertBatches)[i], dispatcher)
		}(i)
	}
}

func handlePanic() {
    r := recover()
    if r != nil {
        fmt.Println("Recovered from panic:", r)

}

    defer handlePanic()
    fmt.Println("Doing something...")

}

func main() {
    fmt.Println("Starting program...")
    doSomething()
    fmt.Println("Program finished.")
}

  1. Współbieżność

Współbieżne przesyłanie plików dalej

package main

import (
    "bufio"
    "fmt"
    "os"
    "sync"
)

type Data struct {
    value int
}

func processFile(filePath string, wg *sync.WaitGroup) {
    defer wg.Done()

    file, err := os.Open(filePath)
    if err != nil {
        fmt.Println("Error opening file:", err)
        return
    }
    defer file.Close()

    scanner := bufio.NewScanner(file)
    for scanner.Scan() {
        line := scanner.Text()
        data := &Data{value: line}
        sendData(data)
    }

    if err := scanner.Err(); err != nil {
        fmt.Println("Error reading file:", err)
    }
}

func sendData(data *Data) {
    // kod umyślnie pominięty 😊
    fmt.Printf("Data sent: %+v\n", data)
}

func main() {
    var wg sync.WaitGroup

    filePaths := []string{"data1.txt", "data2.txt", "data3.txt"}
    for _, filePath := range filePaths {
        wg.Add(1)
        go processFile(filePath, &wg) // cała współbieżność
    }

    wg.Wait()
    fmt.Println("All files processed.")
}

Kanały 🤔
Komunikacja bez pamięci współdzielonej

package main

import (
	"fmt"
)

func sum(a []int, c chan int) {
	sum := 0
	for _, v := range a {
		sum += v
	}
	c <- sum // send sum to c
}

func main() {
	a := []int{7, 2, 8, -9, 4, 0}

	c := make(chan int)
	go sum(a[:len(a)/2], c)
	go sum(a[len(a)/2:], c)
	x, y := <-c, <-c // receive from c

	fmt.Println(x, y, x+y)
}
